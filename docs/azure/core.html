101    def _build_pipeline(
102        self,
103        config: Configuration[HTTPRequestType, HTTPResponseType],
104        *,
105        transport: Optional[HttpTransport[HTTPRequestType, HTTPResponseType]] = None,
106        policies=None,
107        per_call_policies=None,
108        per_retry_policies=None,
109        **kwargs,
110    ) -> Pipeline[HTTPRequestType, HTTPResponseType]:
111        per_call_policies = per_call_policies or []
112        per_retry_policies = per_retry_policies or []
113
114        if policies is None:  # [] is a valid policy list
115            policies = [
116                config.request_id_policy or RequestIdPolicy(**kwargs),
117                config.headers_policy,
118                config.user_agent_policy,
119                config.proxy_policy,
120                ContentDecodePolicy(**kwargs),
121            ]
122            if isinstance(per_call_policies, Iterable):
123                policies.extend(per_call_policies)
124            else:
125                policies.append(per_call_policies)
126
127            policies.extend(
128                [
129                    config.redirect_policy,
130                    config.retry_policy,
131                    config.authentication_policy,
132                    config.custom_hook_policy,
133                ]
134            )
135            if isinstance(per_retry_policies, Iterable):
136                policies.extend(per_retry_policies)
137            else:
138                policies.append(per_retry_policies)
139
140            policies.extend(
141                [
142                    config.logging_policy,
143                    DistributedTracingPolicy(**kwargs),
144                    (SensitiveHeaderCleanupPolicy(**kwargs) if config.redirect_policy else None),
145                    config.http_logging_policy or HttpLoggingPolicy(**kwargs),
146                ]
147            )
148        else:
149            if isinstance(per_call_policies, Iterable):
150                per_call_policies_list = list(per_call_policies)
151            else:
152                per_call_policies_list = [per_call_policies]
153            per_call_policies_list.extend(policies)
154            policies = per_call_policies_list
155
156            if isinstance(per_retry_policies, Iterable):
157                per_retry_policies_list = list(per_retry_policies)
158            else:
159                per_retry_policies_list = [per_retry_policies]
160            if len(per_retry_policies_list) > 0:
161                index_of_retry = -1
162                for index, policy in enumerate(policies):
163                    if isinstance(policy, RetryPolicy):
164                        index_of_retry = index
165                if index_of_retry == -1:
166                    raise ValueError(
167                        "Failed to add per_retry_policies; no RetryPolicy found in the supplied list of policies. "
168                    )
169                policies_1 = policies[: index_of_retry + 1]
170                policies_2 = policies[index_of_retry + 1 :]
171                policies_1.extend(per_retry_policies_list)
172                policies_1.extend(policies_2)
173                policies = policies_1
174
175        if transport is None:
176            # Use private import for better typing, mypy and pyright don't like PEP562
177            from .pipeline.transport._requests_basic import RequestsTransport
178
179            transport = RequestsTransport(**kwargs)
180
181        return Pipeline(transport, policies)

31class CaseInsensitiveEnumMeta(EnumMeta):
32    """Enum metaclass to allow for interoperability with case-insensitive strings.
33
34    Consuming this metaclass in an SDK should be done in the following manner:
35
36    .. code-block:: python

38        from enum import Enum
39        from azure.core import CaseInsensitiveEnumMeta
40
41        class MyCustomEnum(str, Enum, metaclass=CaseInsensitiveEnumMeta):
42            FOO = 'foo'
43            BAR = 'bar'
44
45    """
46
47    def __getitem__(cls, name: str) -> Any:
48        # disabling pylint bc of pylint bug https://github.com/PyCQA/astroid/issues/713
49        return super(CaseInsensitiveEnumMeta, cls).__getitem__(name.upper())
50
51    def __getattr__(cls, name: str) -> Enum:
52        """Return the enum member matching `name`.
53
54        We use __getattr__ instead of descriptors or inserting into the enum
55        class' __dict__ in order to support `name` and `value` being both
56        properties for enum members (which live in the class' __dict__) and
57        enum members themselves.
58
59        :param str name: The name of the enum member to retrieve.
60        :rtype: ~azure.core.CaseInsensitiveEnumMeta
61        :return: The enum member matching `name`.
62        :raises AttributeError: If `name` is not a valid enum member.
63        """
64        try:
65            return cls._member_map_[name.upper()]
66        except KeyError as err:
67            raise AttributeError(name) from err
